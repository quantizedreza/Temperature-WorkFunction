    {\Huge \textbf{Reza Rahemi's Variation In Electron Work Function With Temperature}}

    \vspace{1.5cm}

    {\LARGE $\varphi(T)=\varphi_0-\gamma\frac{(k_BT)^2}{\varphi_0}$}

    \vspace{1cm}

    Where:
    \begin{itemize}
        \item $\gamma$ = Calculated based on Crystal Structure (e.g. FCC, BCC) 
        \item $\varphi_0$= Work Function at T=0, expressed as $\varphi_0 = \frac{e^3 m^{1/2} n^{1/6}}{16^3 \sqrt{3} \, \pi^{5/3} \, \hbar \, \varepsilon_0^{3/2}}$. 

    \end{itemize}

    \vspace{2cm}
    {\Large Scripta Materialia 99, 41-44}
